[BOJ] 2610 회의준비

By Jun Hwi Gu
Posted 2021-09-28 3 min read

KOI 2004 M3/H3

이 문제는 우선 위원회를 분류하기 위해 DFS/BFS 알고리즘을 이용하고, 위원회에서 모든 정점으로 가는 거리(의사전달시간)가 가장 짧은 사람을 대표로 선택하는 문제입니다. 거리는 여러가지 방법이 있지만, N의 범위가 100 이하로 매우 작기 때문에 제 코드에서는 플로이드-와샬 알고리즘을 사용하였습니다.

추신: 문제의 출력은 정렬되어야합니다.

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int INF = 101;
int N, M;
vector<vector<int>> node;
vector<bool> visited;

void DFS(int idx, vector<int>& out) {
	visited[idx] = true;
	out.push_back(idx);
	for (size_t i = 1; i <= N; i++)
		if (!visited[i] && node[idx][i] == 1)
			DFS(i, out);
}

int main() {

	cin >> N >> M;
	visited.resize(N + 1);
    
    // 플로이드-와샬 초기화
	node.resize(N + 1, vector<int>(N + 1, INF));
	for (size_t i = 1; i <= N; i++)
		node[i][i] = 0;

	int a, b;
	for (size_t i = 0; i < M; i++) {
		cin >> a >> b;
		node[a][b] = node[b][a] = 1;
	}

	// 위원회와 위원회원들
	vector<vector<int>> graph;

	for (size_t i = 1; i <= N; i++) {
		if (visited[i]) continue;
		graph.push_back({});
		DFS(i, graph.back());
	}

    // 플로이드-와샬 구현
	for (size_t k = 1; k <= N; k++)
		for (size_t i = 1; i <= N; i++)
			for (size_t j = 1; j <= N; j++)
				if (node[i][j] > node[i][k] + node[k][j])
					node[i][j] = node[i][k] + node[k][j];

	
    // 결과 계산
	vector<int> result(graph.size());
	for (size_t i = 0; i < graph.size(); i++) {
		int val = N, target = 0;
		for (size_t j = 0; j < graph[i].size(); j++) {
      
      		// i번째 위원회의 j번째 위원회원이 갈 수 있는 경로중 최댓값
			int tmp = 0;
			for (size_t k = 1; k <= N; k++)
				if(node[graph[i][j]][k] != INF)
					tmp = max(node[graph[i][j]][k], tmp);
                    
            // 최댓값이 가장 작은 사람이 그 위원회의 대표
			if (tmp < val) {
				val = tmp;
				target = graph[i][j];
			}
		}
		result[i] = target;
	}
    
    // 결과 출력
	cout << graph.size() << endl;
    
	sort(result.begin(), result.end());

	for (size_t i = 0; i < graph.size(); i++)
		cout << result[i] << endl;

}
